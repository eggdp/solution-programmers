// 1. numbers에서 원소를 꺼내야함
// 2. 원소를 꺼내 2배를 한 후 배열에 넣어야한다.
// 인덱스는 0부터 시작numbers [1,2,3,4,5] > numbers[0]=1 
// 원소에 값넣는법 numbers.push(1);
function solution(numbers) { 
    var anwer=[];
    
    let cnt=0;
    while(cnt < numbers.length){
        anwer.push(numbers[cnt] * 2);
        cnt = cnt+1;
    }
    return anwer;
}
